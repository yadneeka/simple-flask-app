#!/bin/bash
set -e

# --- Variables ---
IMAGE_NAME="yadneeka/flask-dummy"
CONTAINER_NAME="${JOB_NAME}-${BUILD_ID}"

# --- Build Docker image ---
echo "Building Docker image..."
docker build -t ${IMAGE_NAME}:${BUILD_ID} .

# --- Run container ---
echo "Running container..."
docker run -d -p 5000:5000 --name ${CONTAINER_NAME} ${IMAGE_NAME}:${BUILD_ID}

# --- List running containers ---
docker ps

# --- Stop and remove container (for cleanup) ---
docker stop ${CONTAINER_NAME}
docker rm ${CONTAINER_NAME}

echo "âœ… Docker build and container run completed successfully."
